  request: NextRequest,
  userCurrency?: string,
  userCountry?: string

function detectBrowserLanguage(request: NextRequest): string | null {
  const acceptLanguage = request.headers.get('accept-language');
  if (!acceptLanguage) return null;
  
  // Parse: en-US,en;q=0.9,de;q=0.8
  const languages = acceptLanguage
    .split(',')
    .map(lang => {
      const [code, q = '1'] = lang.trim().split(';q=');
      return { code: code.trim(), q: parseFloat(q) };
    })
    .sort((a, b) => b.q - a.q);
  
  // Suche nach unterst√ºtzter Sprache
  for (const lang of languages) {
    const locale = BROWSER_LANGUAGE_MAP[lang.code];
    if (locale) return locale;
  }
  
  return null;
}
